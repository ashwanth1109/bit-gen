import React from 'react';

export type TIUI_NAME_PASCAL_CASEProps = {
  /**
   * whether button to increment or decrement should be shown
   */
  shouldIncrement: boolean;
};

export function TIUI_NAME_PASCAL_CASE({ shouldIncrement }: TIUI_NAME_PASCAL_CASEProps) {
  const [count, setCount] = React.useState(0);

  const handleIncrement = () => {
    setCount(count + 1);
  };

  const handleDecrement = () => {
    setCount(count - 1);
  };

  return (
    <>
      <h1>Count: {count}</h1>
      {shouldIncrement ? (
        <button onClick={handleIncrement} role="button">
          Increase
        </button>
      ) : (
        <button onClick={handleDecrement} role="button">
          Decrease
        </button>
      )}
    </>
  );
}
